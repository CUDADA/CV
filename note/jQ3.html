jQuery 操作

css方法

css方法用于css操作与获取

//修改单个样式
//name:样式名 value:样式值
$("li")
      .css("backgroundColor", "pink")
      .css("color", "red")
      .css("fontSize", "32px");
//css(obj)
//修改多个样式
$("li").css({
      backgroundColor:"pink",
      color: "red",
      fontSize:"32px",
      border: "1px solid black"
    });
//获取样式
//css(name)
//name:想要获取的样式
    $("li").eq(0).css("fontSize", "20px");
    $("li").eq(1).css("fontSize", "21px");
    $("li").eq(2).css("fontSize", "22px");
    $("li").eq(3).css("fontSize", "23px");
